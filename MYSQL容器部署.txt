1.创建网络(后续容器间通信需要):
docker network create blog_network

2.启动 MySQL 容器:
docker run -d \
  --name blog_mysql \
  --network blog_network \
  -e MYSQL_ROOT_PASSWORD=davidwang \
  -e MYSQL_DATABASE=blog \
  -p 3306:3306 \
  -v mysql_data:/var/lib/mysql \
  mysql:8.0

3. 检查 MySQL 容器状态:
docker ps
# 查看日志确保 MySQL 已经启动成功
docker logs blog_mysql

4. 导入数据:
# 复制 SQL 文件到容器
docker cp /home/blog_demo/blog_backup.sql blog_mysql:/tmp/

# 进入容器
docker exec -it blog_mysql bash

# 在容器内导入数据
mysql -u root -p blog < /tmp/blog_backup.sql
# 输入密码: davidwang

# 验证数据导入
mysql -u root -p
# 输入密码: davidwang
mysql> use blog;
mysql> show tables;
mysql> select * from User limit 1;
# 确认数据存在后退出
mysql> exit
exit

5. 验证 MySQL 连接:
# 测试从主机连接 MySQL
mysql -h localhost -P 3306 -u root -p
# 输入密码: davidwang
